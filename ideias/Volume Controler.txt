  const $volumeBar = $('#volume-bar');

    // Atualiza o texto com a porcentagem
    const volumePercent = Math.round(volumeValue * 100);
    $volumeText.text(`${volumePercent}%`);

    // Atualiza o Ã­cone de volume
    if (volumeValue === 0) {
