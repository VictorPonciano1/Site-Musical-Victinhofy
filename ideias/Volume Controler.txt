    <button id="muteBtn">游댉</button>

 #muteBtn {
      font-size: 16px;

 }

 SCRIPT PARA O VOLUME
const audios = $('.meuAudio'); // Seleciona todos os elementos de 치udio

      // Controle de volume com slider (afeta todos os players de 치udio)
      $('#controleVolume').on('input', function() {
        const volume = $(this).val();

        // Aplica o volume em todos os 치udios
        audios.each(function() {
          this.volume = volume;
        });

        // Se o volume for zero, todos ficam com mute
        if (volume == 0) {
          audios.each(function() {
            this.muted = true;
          });
          $('#muteBtn').text('游댆');
        } else {
          $('#muteBtn').text('游댉');
        }
      });

      // Mute/Unmute no clique do bot칚o (afeta todos os players de 치udio)
      $('#muteBtn').click(function() {
        toggleMute();
      });

      // Atalho de teclado: tecla "M" (afeta todos os players de 치udio)
      $(document).on('keydown', function(e) {
        // Ignora se estiver digitando em inputs ou textarea
        if ($(e.target).is('input, textarea')) return;

        if (e.key.toLowerCase() === 'm') {
          toggleMute();
        }
      });

      // Fun칞칚o que alterna mute/unmute (afeta todos os players de 치udio)
      function toggleMute() {
        const muteStatus = audios[0].muted; // Checa o estado de mute do primeiro 치udio
        audios.each(function() {
          this.muted = !muteStatus;
        });

        $('#muteBtn').text(muteStatus ? '游댉' : '游댆');
      }
