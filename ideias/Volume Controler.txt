 SCRIPT PARA O VOLUME
  const $volumeBar = $('#volume-bar');
  const $audioPlayer = $('#audio-player')[0]; // jQuery retorna objeto, precisamos do DOM real
  const $volumeText = $('#volume-text');
  const $volumeIcon = $('#volume-icon');

  // Definindo volume inicial
  $audioPlayer.volume = $volumeBar.val();

  // Evento de mudança na barra de volume
  $volumeBar.on('input change', function () {
    const volumeValue = parseFloat($(this).val());
    $audioPlayer.volume = volumeValue;

    // Atualiza o texto com a porcentagem
    const volumePercent = Math.round(volumeValue * 100);
    $volumeText.text(`${volumePercent}%`);

    // Atualiza o ícone de volume
    if (volumeValue === 0) {
      $volumeIcon.text('🔇');
    } else if (volumeValue <= 0.5) {
      $volumeIcon.text('🔉');
    } else {
      $volumeIcon.text('🔊');
    }
  });
